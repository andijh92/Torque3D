104  void* lock(U32 vertexStart, U32 vertexEnd)
105  {
106  if(vertexEnd == 0)
107  vertexEnd = getPointer()->mNumVerts;
108  AssertFatal(vertexEnd > vertexStart, "Can't get a lock with the end before the start.");
109  AssertFatal(vertexEnd <= getPointer()->mNumVerts || getPointer()->mBufferType == GFXBufferTypeVolatile, "Tried to get vertices beyond the end of the buffer!");
110  getPointer()->lock(vertexStart, vertexEnd, &getPointer()->lockedVertexPtr);
111  return getPointer()->lockedVertexPtr;
112  }
113 
114  void unlock()
115  {
116  getPointer()->unlock();
117  }
