61  // Determine the material transparency
62  template<class T> void resolveTransparency(const T shader, F32* dst)
63  {
64  // Start out by getting the <transparency> value
65  *dst = 1.0f;
66  resolveFloat(shader->getTransparency(), dst);
67 
68  // Multiply the transparency by the transparent color
69  ColorF transColor(1.0f, 1.0f, 1.0f, 1.0f);
70  if (shader->getTransparent() && shader->getTransparent()->getColor()) {
71  const domCommon_color_or_texture_type::domColor* color = shader->getTransparent()->getColor();
72  transColor.set(color->getValue()[0], color->getValue()[1], color->getValue()[2], color->getValue()[3]);
73  }
74 
75  if (!shader->getTransparent() || (shader->getTransparent()->getOpaque() == FX_OPAQUE_ENUM_A_ONE)) {
76  // multiply by alpha value and invert (so 1.0 is fully opaque)
77  *dst = 1.0f - (*dst * transColor.alpha);
78  }
79  else {
80  // multiply by average of the RGB values
81  F32 avg = (transColor.red + transColor.blue + transColor.green) / 3;
82  *dst *= avg;
83  }
84  }
