1175 inline void GFXDevice::setVertexBuffer( GFXVertexBuffer *buffer, U32 stream, U32 frequency )
1176 {
1177  AssertFatal( stream < VERTEX_STREAM_COUNT, "GFXDevice::setVertexBuffer - Bad stream index!" );
1178 
1179  if ( buffer && stream == 0 )
1180  setVertexFormat( &buffer->mVertexFormat );
