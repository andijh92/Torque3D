174  S32 getBestCollision(Point3F upVector)
175  {
176  S32 bestCollision = -1;
177 
178  F32 bestAngle = 360.f;
179  S32 count = mCollisionList.getCount();
180  for (U32 i = 0; i < count; ++i)
181  {
182  F32 angle = mRadToDeg(mAcos(mDot(mCollisionList[i].normal, upVector)));
183 
184  if (angle < bestAngle)
185  {
186  bestCollision = i;
187  bestAngle = angle;
188  }
189  }
190 
191  return bestCollision;
192  }
