63  virtual bool visit( const TamlParser& parser, TamlVisitor::PropertyState& propertyState )
64  {
65  // Debug Profiling.
66  PROFILE_SCOPE(TamlAssetReferencedVisitor_Visit);
67 
68  // Fetch property value.
69  const char* pPropertyValue = propertyState.getPropertyValue();
70 
71  // Fetch property word count.
72  const U32 propertyWordCount = StringUnit::getUnitCount( pPropertyValue, ASSET_ASSIGNMENT_TOKEN );
73 
74  // Finish if there's not two words.
75  if ( propertyWordCount != 2 )
76  return true;
77 
78  // Finish if the first word is not an asset signature.
79  if ( StringTable->insert( StringUnit::getUnit( pPropertyValue, 0, ASSET_ASSIGNMENT_TOKEN ) ) != assetLooseIdSignature )
80  return true;
81 
82  // Get asset Id.
83  typeAssetId assetId = StringTable->insert( StringUnit::getUnit( pPropertyValue, 1, ASSET_ASSIGNMENT_TOKEN ) );
84 
85  // Finish if we already have this asset Id.
86  if ( mAssetReferenced.contains( assetId ) )
87  return true;
88 
89  // Insert asset reference.
90  mAssetReferenced.insert( assetId, StringTable->EmptyString() );
91 
92  return true;
93  }
