118 template<class T, dsize_t mapLength>
119 inline void Swizzle<T, mapLength>::ToBuffer( void *destination, const void *source, const dsize_t size ) const
120 {
121  // TODO: OpenMP?
122  AssertFatal( size % ( sizeof( T ) * mapLength ) == 0, "Bad buffer size for swizzle, see docs." );
123  if (!destination || !source) return;
124 
125  T *dest = reinterpret_cast<T *>( destination );
126  const T *src = reinterpret_cast<const T *>( source );
127 
128  for( S32 i = 0; i < size / ( mapLength * sizeof( T ) ); i++ )
129  {
130  for( S32 j = 0; j < mapLength; j++ )
131  *dest++ = src[mMap[j]];
132 
133  src += mapLength;
134  }
135 }
