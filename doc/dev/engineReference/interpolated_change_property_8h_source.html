139 template< typename T, typename TimeSource >
140 void InterpolatedChangeProperty< T, TimeSource >::setTargetValue( const T& value )
141 {
142  if( mTargetValue == value )
143  return;
144 
145  if( mBlendPhaseTime == 0 )
146  {
147  mTargetValue = value;
148  mCurrentValue = value;
149  }
150  else
151  {
152  // Set the source value to the current value (which may be interpolated)
153  // and then start a transition to the given target.
154 
155  mSourceValue = getCurrentValue();
156  mTargetValue = value;
157  mTransitionStartTime = mTimeSource.getPosition();
158  }
159 }

163 template< typename T, typename TimeSource >
164 void InterpolatedChangeProperty< T, TimeSource >::_update() const
165 {
166  // Nothing to do if no transition in progress.
167 
168  if( !mTransitionStartTime )
169  return;
170 
171  // See if we have finished the transition.
172 
173  TimeType deltaTime = mTimeSource.getPosition() - mTransitionStartTime;
174  if( deltaTime >= mBlendPhaseTime )
175  {
176  // We're done.
177  mCurrentValue = mTargetValue;
178  mTransitionStartTime = 0;
179 
180  return;
181  }
182 
183  // Determine the interpolated value.
184 
185  F32 blendFactor = F32( deltaTime ) / F32( mBlendPhaseTime );
186  blendFactor = mTransitionCurve.getUnitValue( blendFactor );
187 
188  mCurrentValue.interpolate( mSourceValue, mTargetValue, blendFactor );
189 }
