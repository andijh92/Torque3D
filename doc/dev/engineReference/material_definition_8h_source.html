147  inline GFXTextureObject* getTex( const FeatureType &type ) const
148  {
149  TextureTable::ConstIterator iter = mTextures.find( &type );

157  inline void setTex( const FeatureType &type, GFXTextureObject *tex )
158  {
159  if ( !tex )
160  {
161  TextureTable::Iterator iter = mTextures.find( &type );
162  if ( iter != mTextures.end() )
163  mTextures.erase( iter );
164 
165  return;
166  }
167 
168  TextureTable::Iterator iter = mTextures.findOrInsert( &type );
169  iter->value = tex;
170  }
171 
174  inline bool hasValidTex() const
175  {

177  for (; iter != mTextures.end(); ++iter)

180  return true;
181  }
182 
183  return false;
184  }
