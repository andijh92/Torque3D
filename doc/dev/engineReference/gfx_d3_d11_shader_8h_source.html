147 const U32 SIO_END = 0x0000FFFF;
148 const U32 SI_OPCODE_MASK = 0x0000FFFF;
149 const U32 SI_COMMENTSIZE_MASK = 0x7FFF0000;
150 const U32 CTAB_CONSTANT = 0x42415443;
151 
152 // Member functions
153 inline bool ConstantTable::Create(const void* data)
154 {
155  const U32* ptr = static_cast<const U32*>(data);
156  while(*++ptr != SIO_END)
157  {
158  if((*ptr & SI_OPCODE_MASK) == SIO_COMMENT)
159  {
160  // Check for CTAB comment
161  U32 comment_size = (*ptr & SI_COMMENTSIZE_MASK) >> 16;
162  if(*(ptr+1) != CTAB_CONSTANT)
163  {
164  ptr += comment_size;
165  continue;
166  }
167 
168  // Read header
169  const char* ctab = reinterpret_cast<const char*>(ptr+2);
170  size_t ctab_size = (comment_size-1)*4;
171 
172  const CTHeader* header = reinterpret_cast<const CTHeader*>(ctab);
173  if(ctab_size < sizeof(*header) || header->Size != sizeof(*header))
174  return false;
175  m_creator = ctab + header->Creator;
176 
177  // Read constants
178  m_constants.reserve(header->Constants);
179  const CTInfo* info = reinterpret_cast<const CTInfo*>(ctab + header->ConstantInfo);
180  for(U32 i = 0; i < header->Constants; ++i)
181  {
182  const CTType* type = reinterpret_cast<const CTType*>(ctab + info[i].TypeInfo);
183 
184  // Fill struct
185  ConstantDesc desc;
186  desc.Name = ctab + info[i].Name;
187  desc.RegisterSet = static_cast<REGISTER_TYPE>(info[i].RegisterSet);
188  desc.RegisterIndex = info[i].RegisterIndex;
189  desc.RegisterCount = info[i].RegisterCount;
190  desc.Rows = type->Rows;
191  desc.Class = static_cast<CONST_CLASS>(type->Class);
192  desc.Type = static_cast<CONST_TYPE>(type->Type);
193  desc.Columns = type->Columns;
194  desc.Elements = type->Elements;
195  desc.StructMembers = type->StructMembers;
196  desc.Bytes = 4 * desc.Elements * desc.Rows * desc.Columns;
197  m_constants.push_back(desc);
198  }
199 
200  return true;
201  }
202  }
203  return false;
204 }
