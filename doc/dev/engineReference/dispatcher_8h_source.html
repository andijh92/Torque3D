123  VectorPtr<IMessageListener *> mListeners;
124 
125  MessageQueue() : mQueueName("")

129  bool isEmpty() { return mListeners.size() == 0; }
130 
131  bool dispatchMessage(const char* event, const char* data)
132  {
133  for(VectorPtr<IMessageListener *>::iterator i = mListeners.begin();i != mListeners.end();i++)
134  {
135  if( !(*i)->onMessageReceived(mQueueName, event, data) )
136  return false;
137  }
138  return true;
139  }
140 
141  bool dispatchMessageObject(Message *msg)

143  for(VectorPtr<IMessageListener *>::iterator i = mListeners.begin();i != mListeners.end();i++)
144  {
145  if( !(*i)->onMessageObjectReceived(mQueueName, msg) )
146  return false;
147  }
148  return true;
