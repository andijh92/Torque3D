177 // x is the first value inside the next column, so the divider x-coords
178 // precede x.
179 inline bool GuiFrameSetCtrl::hitVerticalDivider(S32 x, const Point2I &point)
180 {
181  return((point.x >= S32(x - mFramesetDetails.mBorderWidth)) && (point.x < x) && (point.y >= 0) && (point.y < S32(getHeight())));
182 }

184 //-----------------------------------------------------------------------------

187 {
188  return((point.x >= 0) && (point.x < S32(getWidth())) && (point.y >= S32(y - mFramesetDetails.mBorderWidth)) && (point.y < y));
189 }
