93 template<typename ...ArgTs>
94 struct _EngineFunctionDefaultArguments< void(ArgTs...) > : public EngineFunctionDefaultArguments
95 {
96  template<typename T> using DefVST = typename EngineTypeTraits<T>::DefaultArgumentValueStoreType;
97  std::tuple<DefVST<ArgTs> ...> mArgs;
98 private:
99  using SelfType = _EngineFunctionDefaultArguments< void(ArgTs...) >;
100 
101  template<size_t ...> struct Seq {};
102  template<size_t N, size_t ...S> struct Gens : Gens<N-1, N-1, S...> {};
103 
104  template<size_t ...I> struct Gens<0, I...>{ typedef Seq<I...> type; };
105 
106  template<typename ...TailTs, size_t ...I>
107  static void copyHelper(std::tuple<DefVST<ArgTs> ...> &args, std::tuple<DefVST<TailTs> ...> &defaultArgs, Seq<I...>) {
108  std::tie(std::get<I + (sizeof...(ArgTs) - sizeof...(TailTs))>(args)...) = defaultArgs;
109  }
110 
111  template<typename ...TailTs> using MaybeSelfEnabled = typename std::enable_if<sizeof...(TailTs) <= sizeof...(ArgTs), decltype(mArgs)>::type;
112 
113  template<typename ...TailTs> static MaybeSelfEnabled<TailTs...> tailInit(TailTs ...tail) {
114  std::tuple<DefVST<ArgTs>...> argsT;
115  std::tuple<DefVST<TailTs>...> tailT = std::make_tuple(tail...);
116  SelfType::copyHelper<TailTs...>(argsT, tailT, typename Gens<sizeof...(TailTs)>::type());
117  return argsT;
118  };
119 
120 public:
121  template<typename ...TailTs> _EngineFunctionDefaultArguments(TailTs ...tail)
122  : EngineFunctionDefaultArguments({sizeof...(TailTs)}), mArgs(SelfType::tailInit(tail...))
123  {}
124 };
