323 inline void SceneContainer::freeObjectRef(SceneObjectRef* trash)
324 {
325  trash->object = NULL;
326  trash->nextInBin = NULL;
327  trash->prevInBin = NULL;
328  trash->nextInObj = mFreeRefPool;
329  mFreeRefPool = trash;
330 }

334 inline SceneObjectRef* SceneContainer::allocateObjectRef()
335 {
336  if( mFreeRefPool == NULL )
337  addRefPoolBlock();
338  AssertFatal( mFreeRefPool!=NULL, "Error, should always have a free reference here!" );
339 
340  SceneObjectRef* ret = mFreeRefPool;
341  mFreeRefPool = mFreeRefPool->nextInObj;
342 
343  ret->nextInObj = NULL;
344  return ret;
345 }
