316 inline void SceneContainer::freeObjectRef(SceneObjectRef* trash)
317 {
318  trash->object = NULL;
319  trash->nextInBin = NULL;
320  trash->prevInBin = NULL;
321  trash->nextInObj = mFreeRefPool;
322  mFreeRefPool = trash;
323 }

327 inline SceneObjectRef* SceneContainer::allocateObjectRef()
328 {
329  if( mFreeRefPool == NULL )
330  addRefPoolBlock();
331  AssertFatal( mFreeRefPool!=NULL, "Error, should always have a free reference here!" );
332 
333  SceneObjectRef* ret = mFreeRefPool;
334  mFreeRefPool = mFreeRefPool->nextInObj;
335 
336  ret->nextInObj = NULL;
337  return ret;
338 }
