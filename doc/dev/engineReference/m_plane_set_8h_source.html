220 template< typename T >
221 template< typename P >
222 inline OverlapTestResult PlaneSet< T >::_testOverlap( const P& bounds ) const
223 {
224  bool allInside = true;
225 
226  // First, find out whether there is any plane for which the bounds completely
227  // lie on the negative side. If so, the bounds are clearly outside of the volume.
228 
229  const U32 numPlanes = getNumPlanes();
230  for( U32 nplane = 0; nplane < numPlanes; ++ nplane )
231  {
232  const PlaneF& plane = getPlanes()[ nplane ];
233 
234  const PlaneF::Side side = plane.whichSide( bounds );
235  if( side == PlaneF::Back )
236  return GeometryOutside;
237 
238  if( side != PlaneF::Front )
239  allInside = false;
240  }
241 
242  // Test for containment.
243 
244  if( allInside )
245  return GeometryInside;
246 
247  // Otherwise classify as intersecting.
248 
249  return GeometryIntersecting;
250 }

296 template< typename T >
297 U32 PlaneSet< T >::testPlanes( const Box3F& bounds, U32 planeMask, F32 expand ) const
298 {
299  AssertFatal( mNumPlanes <= 32, "PlaneSet::testPlanes - Too many planes in set!" );
300 
301  // This is based on the paper "A Faster Overlap Test for a Plane and a Bounding Box"
302  // by Kenny Hoff. See http://www.cs.unc.edu/~hoff/research/vfculler/boxplane.html
303 
304  U32 retMask = 0;
305 
306  const U32 numPlanes = getMin( mNumPlanes, U32( 32 ) );
307  for ( S32 i = 0; i < numPlanes; i++ )
308  {
309  U32 mask = ( 1 << i );
310 
311  if ( !( planeMask & mask ) )
312  continue;
313 
314  const PlaneF& plane = mPlanes[ i ];
315 
316  Point3F minPoint, maxPoint;
317 
318  if( plane.x > 0 )
319  {
320  maxPoint.x = bounds.maxExtents.x;
321  minPoint.x = bounds.minExtents.x;
322  }
323  else
324  {
325  maxPoint.x = bounds.minExtents.x;
326  minPoint.x = bounds.maxExtents.x;
327  }
328 
329  if( plane.y > 0 )
330  {
331  maxPoint.y = bounds.maxExtents.y;
332  minPoint.y = bounds.minExtents.y;
333  }
334  else
335  {
336  maxPoint.y = bounds.minExtents.y;
337  minPoint.y = bounds.maxExtents.y;
338  }
339 
340  if( plane.z > 0 )
341  {
342  maxPoint.z = bounds.maxExtents.z;
343  minPoint.z = bounds.minExtents.z;
344  }
345  else
346  {
347  maxPoint.z = bounds.minExtents.z;
348  minPoint.z = bounds.maxExtents.z;
349  }
350 
351  F32 maxDot = mDot( maxPoint, plane );
352 
353  if( maxDot <= - ( plane.d + expand ) )
354  return -1;
355 
356  F32 minDot = mDot( minPoint, plane );
357 
358  if( ( minDot + plane.d ) < 0.0f )
359  retMask |= mask;
360  }
361 
362  return retMask;
363 }

367 template< typename T >
368 bool PlaneSet< T >::clipSegment( Point3F &pnt0, Point3F &pnt1 ) const
369 {
370  F32 tmin = F32_MAX;
371  F32 tmax = -F32_MAX;
372  U32 hitCount = 0;
373  Point3F tpnt;
374 
375  const U32 numPlanes = mNumPlanes;
376  for( U32 i = 0; i < numPlanes; ++ i )
377  {
378  const PlaneF &plane = mPlanes[ i ];
379 
380  F32 t = plane.intersect( pnt0, pnt1 );
381 
382  if( t >= 0.0f && t <= 1.0f )
383  {
384  tpnt.interpolate( pnt0, pnt1, t );
385 
386  if ( isContained( tpnt, 1.0e-004f ) )
387  {
388  tmin = getMin( tmin, t );
389  tmax = getMax( tmax, t );
390  hitCount ++;
391  }
392  }
393  }
394 
395  // If we had no intersections then either both points are inside or both are outside.
396 
397  if( hitCount == 0 )
398  return isContained( pnt0 );
399 
400  // If we had one intersection then we have one point inside.
401  // tmin and tmax are the same here.
402  if( hitCount == 1 )
403  {
404  if( isContained( pnt0 ) )
405  pnt1.interpolate( pnt0, pnt1, tmax );
406  else
407  pnt0.interpolate( pnt0, pnt1, tmin );
408  }
409  else
410  {
411  Point3F prevPnt0( pnt0 );
412  Point3F prevPnt1( pnt1 );
413 
414  if( tmin < F32_MAX )
415  pnt0.interpolate( prevPnt0, prevPnt1, tmin );
416  if( tmax > -F32_MAX )
417  pnt1.interpolate( prevPnt0, prevPnt1, tmax );
418  }
419 
420  return true;
421 }

425 template< typename T >
426 U32 PlaneSet< T >::clipPolygon( const Point3F* inVertices, U32 inNumVertices, Point3F* outVertices, U32 maxOutVertices ) const
427 {
428  TempAlloc< Point3F > tempBuffer( inNumVertices + mNumPlanes );
429 
430  // We use two buffers as interchanging roles as source and target.
431  // For the first iteration, inVertices is the source.
432 
433  Point3F* tempPolygon = tempBuffer;
434  Point3F* clippedPolygon = const_cast< Point3F* >( inVertices );
435 
436  U32 numClippedPolygonVertices = inNumVertices;
437  U32 numTempPolygonVertices = 0;
438 
439  for( U32 nplane = 0; nplane < mNumPlanes; ++ nplane )
440  {
441  // Make the output of the last iteration the
442  // input of this iteration.
443 
444  swap( tempPolygon, clippedPolygon );
445  numTempPolygonVertices = numClippedPolygonVertices;
446 
447  if( maxOutVertices < numTempPolygonVertices + 1 )
448  return 0;
449 
450  // Clip our current remainder of the original polygon
451  // against the current plane.
452 
453  const PlaneF& plane = mPlanes[ nplane ];
454  numClippedPolygonVertices = plane.clipPolygon( tempPolygon, numTempPolygonVertices, clippedPolygon );
455 
456  // If the polygon was completely on the backside of the plane,
457  // then polygon is outside the frustum. In this case, return false
458  // to indicate we haven't clipped anything.
459 
460  if( !numClippedPolygonVertices )
461  return 0;
462 
463  // On first iteration, replace the inVertices with the
464  // outVertices buffer.
465 
466  if( tempPolygon == inVertices )
467  tempPolygon = outVertices;
468  }
469 
470  // If outVertices isn't the target buffer of the last
471  // iteration, copy the vertices over from the temporary
472  // buffer.
473 
474  if( clippedPolygon != outVertices )
475  dMemcpy( outVertices, clippedPolygon, numClippedPolygonVertices * sizeof( Point3F ) );
476 
477  return numClippedPolygonVertices;
478 }
