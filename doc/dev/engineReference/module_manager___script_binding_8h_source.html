136 DefineEngineMethod(ModuleManager, findModules, String, (bool loadedOnly), (false),
137  "Find all the modules registered with the specified loaded state.\n"
138  "@param loadedOnly Whether to return only modules that are loaded or not.\n"
139  "@return A list of space - separated module definition object Ids.\n")
140 {
141  // Find module type definitions.
142  Vector<const ModuleDefinition*> moduleDefinitions;
143 
144  // Find modules.
145  object->findModules( loadedOnly, moduleDefinitions );
146 
147  // Fetch module definition count.
148  const U32 moduleDefinitionCount = (U32)moduleDefinitions.size();
149 
150  // Finish if no module definition were found.
151  if ( moduleDefinitionCount == 0 )
152  return StringTable->EmptyString();
153 
154  // Create a return buffer.
155  S32 bufferSize = 4096;
156  char* pReturnBuffer = Con::getReturnBuffer( bufferSize );
157  char* pBufferWrite = pReturnBuffer;
158 
159  // Iterate module definitions.
160  for ( ModuleManager::typeConstModuleDefinitionVector::const_iterator moduleDefinitionItr = moduleDefinitions.begin(); moduleDefinitionItr != moduleDefinitions.end(); ++moduleDefinitionItr )
161  {
162  // Fetch module definition.
163  const ModuleDefinition* pModuleDefinition = *moduleDefinitionItr;
164 
165  // Format module definition.
166  const U32 offset = dSprintf( pBufferWrite, bufferSize, "%d ", pModuleDefinition->getId() );
167  pBufferWrite += offset;
168  bufferSize -= offset;
169 
170  // Are we out of buffer space?
171  if ( bufferSize <= 0 )
172  {
173  // Yes, so warn.
174  Con::warnf( "ModuleManager::findModules() - Ran out of buffer space." );
175  break;
176  }
177  }
178 
179  return pReturnBuffer;
180 }
