318 void SimSet::findObjectByType( Vector<T*> &foundObjects )
319 {
320  T *curObj;
321  SimSet *curSet;
322 
323  lock();
324 
325  // Loop through our child objects.
326 
327  SimObjectList::iterator itr = objectList.begin();
328 
329  for ( ; itr != objectList.end(); itr++ )
330  {
331  curObj = dynamic_cast<T*>( *itr );
332  curSet = dynamic_cast<SimSet*>( *itr );
333 
334  // If child object is a set, call recursively into it.
335  if ( curSet )
336  curSet->findObjectByType( foundObjects );
337 
338  // Add this child object if appropriate.
339  if ( curObj )
340  foundObjects.push_back( curObj );
341  }
342 
343  // Add this object if appropriate.
344  curObj = dynamic_cast<T*>(this);
345  if ( curObj )
346  foundObjects.push_back( curObj );
347 
348  unlock();
349 }

354  T *curObj;

357  lock();
358 
359  // Loop through our child objects.

362

364  {
365  curObj = dynamic_cast<T*>( *itr );
366  curSet = dynamic_cast<SimSet*>( *itr );
367 
368  // If child object is a set, call recursively into it.
369  if ( curSet )
370  curSet->findObjectByCallback( fn, foundObjects );
371 
372  // Add this child object if appropriate.
373  if ( curObj && ( fn == NULL || fn( curObj ) ) )
374  foundObjects.push_back( curObj );

378  curObj = dynamic_cast<T*>(this);
379  if ( curObj && ( fn == NULL || fn( curObj ) ) )
380  foundObjects.push_back( curObj );
