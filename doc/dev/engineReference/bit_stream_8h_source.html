100  void writeCussedU32(U32 val)
101  {
102  // Is it zero?
103  if(writeFlag(val == 0))
104  return;
105 
106  if(writeFlag(val <= 0xF)) // 4 bit
107  writeRangedU32(val, 0, 0xF);
108  else if(writeFlag(val <= 0xFF)) // 8 bit
109  writeRangedU32(val, 0, 0xFF);
110  else if(writeFlag(val <= 0xFFFF)) // 16 bit
111  writeRangedU32(val, 0, 0xFFFF);
112  else if(writeFlag(val <= 0xFFFFFF)) // 24 bit
113  writeRangedU32(val, 0, 0xFFFFFF);
114  else
115  writeRangedU32(val, 0, 0xFFFFFFFF);
116  }

325 inline bool BitStream::readFlag()
326 {
327  if(bitNum > maxReadBitNum)
328  {
329  error = true;
330  AssertFatal(false, "Out of range read");
331  return false;
332  }
333  S32 mask = 1 << (bitNum & 0x7);
334  bool ret = (*(dataPtr + (bitNum >> 3)) & mask) != 0;
335  bitNum++;
336  return ret;
337 }

339 inline void BitStream::writeRangedU32(U32 value, U32 rangeStart, U32 rangeEnd)
340 {
341  AssertFatal(value >= rangeStart && value <= rangeEnd, "Out of bounds value!");
342  AssertFatal(rangeEnd >= rangeStart, "error, end of range less than start");
343 
344  U32 rangeSize = rangeEnd - rangeStart + 1;
345  U32 rangeBits = getBinLog2(getNextPow2(rangeSize));
346 
347  writeInt(S32(value - rangeStart), S32(rangeBits));
348 }

372 inline void BitStream::writeRangedF32( F32 value, F32 min, F32 max, U32 numBits )
373 {
374  value = ( mClampF( value, min, max ) - min ) / ( max - min );
375  writeInt( (S32)mFloor(value * F32( (1 << numBits) - 1 )), numBits );
376 }

378 inline F32 BitStream::readRangedF32( F32 min, F32 max, U32 numBits )
379 {
380  F32 value = (F32)readInt( numBits );
381  value /= F32( ( 1 << numBits ) - 1 );
382  return min + value * ( max - min );
