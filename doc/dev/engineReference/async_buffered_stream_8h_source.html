159 template< typename T, typename Stream >
160 AsyncBufferedInputStream< T, Stream >::AsyncBufferedInputStream
161  ( const Stream& stream,
162  U32 numSourceElementsToRead,
163  U32 numReadAhead,
164  bool isLooping,
165  ThreadPool* threadPool,
166  ThreadContext* threadContext )
167  : Parent( stream ),
168  mIsLooping( isLooping ),
169  mIsStopped( false ),
170  mNumRemainingSourceElements( numSourceElementsToRead ),
171  mNumBufferedElements( 0 ),
172  mMaxBufferedElements( numReadAhead ),
173  mThreadPool( threadPool ),
174  mThreadContext( threadContext )
175 {
176  if( mIsLooping )
177  {
178  // Stream is looping so we don't count down source elements.
179 
180  mNumRemainingSourceElements = 0;
181  }
182  else if( !mNumRemainingSourceElements )
183  {
184  // If not given number of elements to read, see if the source
185  // stream is sizeable. If so, take its size as the number of elements.

187  if( dynamic_cast< ISizeable<>* >( &Deref( stream ) ) )
188  mNumRemainingSourceElements = ( ( ISizeable<>* ) &Deref( stream ) )->getSize();
189  else

191  // Can't tell how many source elements there are.
192 
193  mNumRemainingSourceElements = U32_MAX;
194  }
195  }

210 template< typename T, typename Stream >
211 void AsyncBufferedInputStream< T, Stream >::_onArrival( const ElementType& element )
212 {
213  mBufferedElements.pushBack( element );
214 
215  // Adjust buffer count.
216 
217  while( 1 )
218  {
219  S32 numBuffered = mNumBufferedElements;
220  if( dCompareAndSwap( mNumBufferedElements, numBuffered, numBuffered + 1 ) )
221  {
222  // If we haven't run against the lookahead limit and haven't reached
223  // the end of the stream, immediately trigger a new request.
224 
225  if( !mIsStopped && ( numBuffered + 1 ) < mMaxBufferedElements )
226  _requestNext();
227 
228  break;
229  }
230  }
231 }

235 template< typename T, typename Stream >
236 U32 AsyncBufferedInputStream< T, Stream >::read( ElementType* buffer, U32 num )
237 {
238  if( !num )
239  return 0;
240 
241  U32 numRead = 0;
242  for( U32 i = 0; i < num; ++ i )
243  {
244  // Try to pop a element off the buffered element list.
245 
246  ElementType element;
247  if( mBufferedElements.tryPopFront( element ) )
248  {
249  buffer[ i ] = element;
250  numRead ++;
251  }
252  else
253  break;
254  }
255 
256  // Get the request chain going again, if it has stopped.
257 
258  while( 1 )
259  {
260  U32 numBuffered = mNumBufferedElements;
261  U32 newNumBuffered = numBuffered - numRead;
262 
263  if( dCompareAndSwap( mNumBufferedElements, numBuffered, newNumBuffered ) )
264  {
265  if( numBuffered == mMaxBufferedElements )
266  _requestNext();
267 
268  break;
269  }
270  }
271 
272  return numRead;
273 }

391 template< typename T, typename Stream, class ReadItem >
392 void AsyncSingleBufferedInputStream< T, Stream, ReadItem >::_requestNext()
393 {
394  Stream& stream = this->getSourceStream();
395  bool isEOS = !this->mNumRemainingSourceElements;
396  if( isEOS && this->mIsLooping )
397  {
398  SourceStreamType* s = &Deref( stream );
399  dynamic_cast< IResettable* >( s )->reset();
400  isEOS = false;
401  }
402  else if( isEOS )
403  return;
404 
405  //TODO: could scale priority depending on feed status
406 
407  // Queue a stream packet work item.
408 
409  if( !this->mIsLooping && this->mNumRemainingSourceElements != U32_MAX )
410  -- this->mNumRemainingSourceElements;
411 
412  ThreadSafeRef< ThreadWorkItem > workItem;
413  _newReadItem( workItem );
414  this->mThreadPool->queueWorkItem( workItem );
415 }
