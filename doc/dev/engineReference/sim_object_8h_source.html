1009

1011 
1012  ~SimObjectPtr() { set((WeakRefBase::WeakReference*)NULL); }
1013 
1014  SimObjectPtr<T>& operator=(const SimObjectPtr ref)
1015  {
1016  set(ref.mReference);
1017  return *this;
1018  }
1019  SimObjectPtr<T>& operator=(T *ptr)
1020  {
1021  set(ptr);
1022  return *this;
1023  }
1024 
1025  protected:
1026  void set(WeakRefBase::WeakReference * ref)
1027  {
1028  if( ref == this->mReference )
1029  return;
1030 
1031  if( this->mReference )
1032  {
1033  // Auto-delete
1034  T* obj = this->getPointer();
1035  if ( this->mReference->getRefCount() == 2 && obj && obj->isAutoDeleted() )
1036  obj->deleteObject();
1037 
1038  this->mReference->decRefCount();
1039  }
1040  this->mReference = NULL;
1041  if( ref )
1042  {
1043  this->mReference = ref;
1044  this->mReference->incRefCount();
1045  }
1046  }
1047 
1048  void set(T * obj)
1049  {
1050  set(obj ? obj->getWeakReference() : (WeakRefBase::WeakReference *)NULL);
1051  }
