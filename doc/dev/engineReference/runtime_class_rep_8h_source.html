108  void consoleRegister()
109  {
110  AssertFatal( !smConRegistered, "Calling consoleRegister, but this type is already linked into the class list" );
111 
112  if( !smConRegistered )
113  registerClassRep( this );
114 
115  // Now initialize the namespace
116  mNamespace = Con::lookupNamespace( StringTable->insert( getClassName() ) );
117  mNamespace->mClassRep = this;
118 
119  // Perform field initialization
120  sg_tempFieldList.setSize(0);
121 
122  init();
123 
124  // So if we have things in it, copy it over...
125  if ( sg_tempFieldList.size() != 0 )
126  mFieldList = sg_tempFieldList;
127 
128  // And of course delete it every round.
129  sg_tempFieldList.clear();
130 
131  smConRegistered = true;
132  }
