185  U32 _getPixelOffset( th_ycbcr_buffer buffer, U32 plane, U32 x, U32 y ) const
186  {
187  switch( getDecoderPixelFormat() )
188  {
189  case PIXEL_FORMAT_444: break;
190  case PIXEL_FORMAT_422: if( plane != 0 ) x >>= 1; break;
191  case PIXEL_FORMAT_420: if( plane != 0 ) { x >>= 1; y >>= 1; } break;
192 
193  default:
194  AssertFatal( false, "OggTheoraDecoder::_getPixelOffset() - invalid pixel format" );
195  }
196 
197  return ( y * buffer[ plane ].stride + x );
198  }
