157 template< typename Polyhedron >
158 void PolyhedronBoxIntersector< Polyhedron >::_preprocess( const MatrixF& objToWorld, const Point3F& scale )
159 {
160  PROFILE_SCOPE( PolyhedronBoxIntersector_preprocess );
161 
162  // Transform the planes.
163 
164  const U32 numPlanes = this->mTester.getNumPlanes();
165  const typename Polyhedron::PlaneType* planes = this->mTester.getPlanes();
166 
167  PlaneTransformer transformer;
168  transformer.set( objToWorld, scale );
169 
170  mPlanes.setSize( numPlanes );
171  for( U32 i = 0; i < numPlanes; ++ i )
172  transformer.transform( planes[ i ], mPlanes[ i ] );
173 
174  // Extract the silhouettes for each of the three
175  // orthographic projections.
176 
177  const U32 numEdges = this->mTester.getNumEdges();
178  const typename Polyhedron::EdgeType* edges = this->mTester.getEdges();
179  const typename Polyhedron::PointType* points = this->mTester.getPoints();
180 
181  for( U32 i = 0; i < 3; ++ i )
182  {
183  U32 numEdgesThisProj = 0;
184 
185  // Gather edge-lines for this projection.
186 
187  for( U32 n = 0; n < numEdges; ++ n )
188  {
189  const typename Polyhedron::EdgeType& edge = edges[ n ];
190 
191  // Compute dot product with face normals. With our projection
192  // pointing straight down the current axis, this is reduced to
193  // '1*normal[i]'.
194 
195  F32 dotFace[ 2 ];
196 
197  dotFace[ 0 ] = mPlanes[ edge.face[ 0 ] ][ i ];
198  dotFace[ 1 ] = mPlanes[ edge.face[ 1 ] ][ i ];
199 
200  // Skip edge if not a silhouette edge in this view.
201 
202  if( mSign( dotFace[ 0 ] ) == mSign( dotFace[ 1 ] ) )
203  continue;
204 
205  // Find out which face is the front facing one. Since we expect normals
206  // to be pointing inwards, this means a reversal of the normal back facing
207  // test and we're looking for a normal facing the *same* way as our projection.
208 
209  const U32 frontFace = dotFace[ 0 ] > 0.f ? 0 : 1;
210  if( dotFace[ frontFace ] <= 0.f )
211  continue; // This face or other face is perpendicular to us.
212 
213  // Now we want to find the line equation for the edge. For that, we first need
214  // the normal. The direction of the normal is important so that we identify
215  // the half-spaces correctly. We want it to be pointing to the inside of the
216  // polyhedron.
217 
218  Point3F v1 = points[ edge.vertex[ 0 ] ];
219  Point3F v2 = points[ edge.vertex[ 1 ] ];
220 
221  v1.convolve( scale );
222  v2.convolve( scale );
223 
224  objToWorld.mulP( v1 );
225  objToWorld.mulP( v2 );
226 
227  Point2F q = _project( i, v1 ); // First point on line.
228  Point2F p = _project( i, v2 ); // Second point on line.
229 
230  if( frontFace != 0 )
231  swap( p, q );
232 
233  Point2F normal( - ( p.y - q.y ), p.x - q.x );
234  normal.normalize();
235 
236  // Now compute c.
237 
238  const F32 c = mDot( - q, normal );
239 
240  // Add the edge.
241 
242  mEdgeLines.push_back(
243  Point3F( normal.x, normal.y, c )
244  );
245 
246  numEdgesThisProj ++;
247  }
248 
249  mNumEdgeLines[ i ] = numEdgesThisProj;
250  }
251 }

255 template< typename Polyhedron >
256 OverlapTestResult PolyhedronBoxIntersector< Polyhedron >::test( const Box3F& box ) const
257 {
258  PROFILE_SCOPE( PolyhedronBoxIntersector_test );
259 
260  // -- Bounding box tests. --
261 
262  // If the box does not intersect with the AABB of the polyhedron,
263  // it must be outside.
264 
265  if( !mBounds.isOverlapped( box ) )
266  return GeometryOutside;
267 
268  // If the polyhedron's bounding box is fully contained in the given box,
269  // the box is intersecting.
270 
271  if( box.isContained( mBounds ) )
272  return GeometryIntersecting;
273 
274  // -- Face-plane tests. --
275 
276  bool insideAll = true;
277 
278  // Test each of the planes to see if the bounding box lies
279  // fully in the negative space of any one of them.
280 
281  const U32 numPlanes = mPlanes.size();
282  for( U32 i = 0; i < numPlanes; ++ i )
283  {
284  const PlaneF& plane = mPlanes[ i ];
285 
286  PlaneF::Side boxSide = plane.whichSide( box );
287  if( boxSide == PlaneF::Back )
288  return GeometryOutside;
289 
290  insideAll &= ( boxSide == PlaneF::Front );
291  }
292 
293  // If the box is on the positive space of all of the polyhedron's
294  // planes, it's inside.
295 
296  if( insideAll )
297  return GeometryInside;
298 
299  // -- Edge-line tests. --
300 
301  U32 edgeLineIndex = 0;
302  for( U32 i = 0; i < 3; ++ i )
303  {
304  // Determine the mapping of 3D to 2D for this projection.
305 
306  U32 xIndex = 0;
307  U32 yIndex = 0;
308 
309  switch( i )
310  {
311  case 0: xIndex = 1; yIndex = 2; break;
312  case 1: xIndex = 0; yIndex = 2; break;
313  case 2: xIndex = 0; yIndex = 1; break;
314  }
315 
316  // Go through the edge-lines for this projection and
317  // test the p-vertex for each edge line.
318 
319  const U32 numEdgesForThisProj = mNumEdgeLines[ i ];
320  for( U32 n = 0; n < numEdgesForThisProj; ++ n, edgeLineIndex ++ )
321  {
322  const Point3F& edgeLine = mEdgeLines[ edgeLineIndex ];
323 
324  // Determine the p-vertex for the current AABB/edge combo.
325  // Need to account for the axis flipping we have applied to maintain
326  // a right-handed coordinate system.
327 
328  Point2F pVertex;
329 
330  switch( i )
331  {
332  case 0:
333  pVertex.x = - ( edgeLine.x < 0.f ? box.maxExtents[ xIndex ] : box.minExtents[ xIndex ] );
334  pVertex.y = edgeLine.y > 0.f ? box.maxExtents[ yIndex ] : box.minExtents[ yIndex ];
335  break;
336 
337  case 1:
338  pVertex.x = edgeLine.x > 0.f ? box.maxExtents[ xIndex ] : box.minExtents[ xIndex ];
339  pVertex.y = edgeLine.y > 0.f ? box.maxExtents[ yIndex ] : box.minExtents[ yIndex ];
340  break;
341 
342  case 2:
343  pVertex.x = edgeLine.x > 0.f ? box.maxExtents[ xIndex ] : box.minExtents[ xIndex ];
344  pVertex.y = - ( edgeLine.y < 0.f ? box.maxExtents[ yIndex ] : box.minExtents[ yIndex ] );
345  break;
346  }
347 
348  // See if the p-vertex lies inside in the negative half-space of the
349  // edge line. If so, the AABB is not intersecting the polyhedron in
350  // this projection so we can conclude our search here.
351 
352  const F32 d = edgeLine.x * pVertex.x + edgeLine.y * pVertex.y + edgeLine.z;
353  if( d < 0.f )
354  return GeometryOutside;
355  }
356  }
357 
358  // Done. Determined to be intersecting.
359 
360  return GeometryIntersecting;
361 }
