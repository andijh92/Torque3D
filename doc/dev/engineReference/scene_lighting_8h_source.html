198  sgSceneLightingProcessEvent(U32 lightIndex, S32 objectIndex, U32 event)
199  {
200  sgLightIndex = lightIndex;
201  sgObjectIndex = objectIndex;
202  sgEvent = event;
203  }
204  void process(SimObject * object)
205  {
206  AssertFatal(object, "SceneLightingProcessEvent:: null event object!");
207  if(!object)
208  return;
209 
210  SceneLighting *sl = static_cast<SceneLighting*>(object);
211  switch(sgEvent)
212  {
213  case sgLightingStartEventType:
214  sl->sgLightingStartEvent();
215  break;
216  case sgLightingCompleteEventType:
217  sl->sgLightingCompleteEvent();
218  break;
219 
220  case sgTGEPassSetupEventType:
221  sl->sgTGEPassSetupEvent();
222  break;
223  case sgTGELightStartEventType:
224  sl->sgTGELightStartEvent(sgLightIndex);
225  break;
226  case sgTGELightProcessEventType:
227  sl->sgTGELightProcessEvent(sgLightIndex, sgObjectIndex);
228  break;
229  case sgTGELightCompleteEventType:
230  sl->sgTGELightCompleteEvent(sgLightIndex);
231  break;
232 
233  case sgSGPassSetupEventType:
234  sl->sgSGPassSetupEvent();
235  break;
236  case sgSGObjectStartEventType:
237  sl->sgSGObjectStartEvent(sgObjectIndex);
238  break;
239  case sgSGObjectProcessEventType:
240  sl->sgSGObjectProcessEvent(sgLightIndex, sgObjectIndex);
241  break;
242  case sgSGObjectCompleteEventType:
243  sl->sgSGObjectCompleteEvent(sgObjectIndex);
244  break;
245 
246  default:
247  return;
248  };
249  };
