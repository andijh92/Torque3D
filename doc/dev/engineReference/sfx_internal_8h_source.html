180 {
181  public:

184 
185  protected:

188  SFXVoice* mVoice;

191  mutable U32 mLastPos;
192 
193  public:
194 
195  SFXVoiceTimeSource( SFXVoice* voice )
196  : mVoice( voice ), mLastPos( 0 ) {}
197 
198  U32 getPosition() const
199  {
200  U32 samplePos = mVoice->_tell();
201 
202  // The device playback cursor may snap back to an undefined value as soon
203  // as all the data has been consumed. However, for us to be a reliable
204  // time source, we can't let that happen so as soon as the device play cursor
205  // goes back to a sample position we have already passed, we start reporting an
206  // end-of-stream position.
207 
208  if( samplePos < mLastPos && mVoice->mBuffer != NULL )
209  samplePos = mVoice->mBuffer->getNumSamples();
210  else
211  mLastPos = samplePos;
212 
213  return samplePos;
214  }
215 };

295 
296  // SFXBuffer.
297  virtual void write( SFXStreamPacket* const* packets, U32 num );

301  U32 getSamplePos( U32 bufferOffset ) const
302  {
303  if( !mBufferSize )
304  return ( bufferOffset / getFormat().getBytesPerSample() );
305 
306  const U32 writeOffset = mWriteOffset; // Concurrent writes on this one.
307  const U32 writeOffsetRelative = writeOffset % mBufferSize;
308 
309  U32 numBufferedBytes;
310  if( !writeOffset )
311  numBufferedBytes = 0;
312  else if( writeOffsetRelative > bufferOffset )
313  numBufferedBytes = writeOffsetRelative - bufferOffset;
314  else
315  // Wrap-around.
316  numBufferedBytes = mBufferSize - bufferOffset + writeOffsetRelative;
317 
318  const U32 bytePos = writeOffset - numBufferedBytes;
319 
320  return ( bytePos / getFormat().getBytesPerSample() );
321  }
322 
323  public:
324

326  SFXWrapAroundBuffer( SFXDescription* description )
327  : Parent( description ), mBufferSize( 0 ) {}
