159 struct Edge
160 {
161  Point2F normal;
162  Point2I position;
163  Point2I extent;
164  F32 margin;
165 
166  Edge(): normal(0.f,0.f),
167  position(0,0),
168  extent(0,0),
169  margin(1.f){};
170  Edge( const Point2I &inPoint, const Point2F &inNormal )
171  {
172  normal = inNormal;
173  margin = 2.f;
174 
175  if( normal.x == 1.f || normal.x == -1.f )
176  {
177  // Vertical Edge
178  position.x = inPoint.x;
179  position.y = 0;
180 
181  extent.x = 1;
182  extent.y = 1;
183  }
184  else if( normal.y == 1.f || normal.y == -1.f )
185  {
186  // Horizontal Edge
187  position.y = inPoint.y;
188  position.x = 0;
189 
190  extent.x = 1;
191  extent.y = 1;
192  }
193  else
194  AssertFatal(false,"Edge point constructor cannot construct an Edge without an axis-aligned normal.");
195  }
196 
197  // Copy Constructor
198  Edge( const Edge &inEdge )
199  {
200  normal = inEdge.normal;
201  position = inEdge.position;
202  extent = inEdge.extent;
203  margin = inEdge.margin;
204  }
205 
206  // RectI cast operator overload
207  operator const RectI() const
208  {
209  if( normal.x == 1.f || normal.x == -1.f )
210  {
211  // Vertical Edge
212  RectI retRect = RectI( position.x, position.y, 1, position.y + extent.y );
213  // Expand Rect by Margin along the X Axis
214  retRect.inset(-margin,0);
215 
216  return retRect;
217  }
218  else if( normal.y == 1.f || normal.y == -1.f )
219  {
220  // Horizontal Edge
221  RectI retRect = RectI( position.x, position.y , position.x + extent.x, 1 );
222  // Expand Rect by Margin along the Y Axis
223  retRect.inset(0,-margin);
224  return retRect;
225  }
226 
227  // CodeReview this code only deals with axis-aligned edges [6/8/2007 justind]
228  AssertFatal(false,"Edge cast operator cannot construct a Rect from an Edge that is not axis-aligned.");
229  return RectI( 0,0,0,0 );
230  }
231 
232  inline bool hit( const Edge &inEdge ) const
233  {
234  const RectI thisRect = *this;
235  const RectI thatRect = inEdge;
236 
237  return thisRect.overlaps( thatRect );
238  }
239 };

243 struct EdgeRectI
244 {
245  Edge left;
246  Edge top;
247  Edge bottom;
248  Edge right;
249 
250  EdgeRectI(){ }
251 
252  EdgeRectI( const RectI &inRect, F32 inMargin )
253  {
254  // Left Edge
255  left.normal = Point2F( -1.f, 0.f );
256  left.position.x= inRect.point.x;
257  left.position.y= 0;
258  left.extent = Point2I(inRect.point.y, inRect.point.y + inRect.extent.y);
259  left.margin = inMargin;
260 
261  // Right Edge
262  right.normal = Point2F( 1.f, 0.f );
263  right.position.x = inRect.point.x + inRect.extent.x;
264  right.position.y = 0;
265  right.extent = Point2I(inRect.point.y, inRect.point.y + inRect.extent.y);
266  right.margin = inMargin;
267 
268  // Top Edge
269  top.normal = Point2F( 0.f, 1.f );
270  top.position.y = inRect.point.y;
271  top.position.x = 0;
272  top.extent = Point2I(inRect.point.x + inRect.extent.x, inRect.point.x);
273  top.margin = inMargin;
274 
275  // Bottom Edge
276  bottom.normal = Point2F( 0.f, -1.f );
277  bottom.position.y= inRect.point.y + inRect.extent.y;
278  bottom.position.x=0;
279  bottom.extent = Point2I(inRect.point.x + inRect.extent.x, inRect.point.x);
280  bottom.margin = inMargin;
281  }
282 
283  // Copy constructor
284  EdgeRectI( const EdgeRectI &inEdgeRect )
285  {
286  left = inEdgeRect.left;
287  right = inEdgeRect.right;
288  top = inEdgeRect.top;
289  bottom = inEdgeRect.bottom;
