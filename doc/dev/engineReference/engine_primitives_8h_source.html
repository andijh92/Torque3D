51 // String is special in the way its data is exchanged through the API. Through
52 // calls, strings are passed as plain, null-terminated UTF-16 character strings.
53 // In addition, strings passed back as return values from engine API functions
54 // are considered to be owned by the API layer itself. The rule here is that such
55 // a string is only valid until the next API call is made. Usually, control layers
56 // will immediately copy and convert strings to their own string type.
57 _DECLARE_TYPE_R(String);
58 template<>
59 struct EngineTypeTraits< String > : public _EnginePrimitiveTypeTraits< String >
60 {
61  typedef const UTF16* ArgumentValueType;
62  typedef const UTF16* ReturnValueType;
63 
64  //FIXME: this needs to be sorted out; for now, we store default value literals in ASCII
65  typedef const char* DefaultArgumentValueStoreType;
66 
67  static const UTF16* ReturnValue( const String& str )
68  {
69  static String sTemp;
70  sTemp = str;
71  return sTemp.utf16();
72  }
73 };
