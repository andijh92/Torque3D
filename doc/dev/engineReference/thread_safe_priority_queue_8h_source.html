199 template< typename T, typename K, bool SORT_MIN_TO_MAX, U32 MAX_LEVEL, U32 PROBABILISTIC_BIAS >
200 void ThreadSafePriorityQueue< T, K, SORT_MIN_TO_MAX, MAX_LEVEL, PROBABILISTIC_BIAS >::insert( KeyType priority, const ValueType& value, NodePtr& outResult )
201 {
202  // Create a new node at a random level.
203 
204  outResult = NULL;
205  NodePtr::unsafeWrite( outResult, new Node( priority, value ) );
206  U32 resultNodeLevel = outResult->getLevel();
207 
208  // Link up all the levels. Do this bottom-up instead of
209  // top-down (as would be the right way for a skiplist) so
210  // that our list state always remains valid. If going top-down,
211  // we'll insert nodes with NULL pointers at their lower levels.
212 
213  U32 currentLevel = 0;
214  do
215  {
216  while( 1 )
217  {
218  NodePtr nextNode;
219  NodePtr prevNode;
220 
221  scanFromHead( prevNode, nextNode, currentLevel, priority );
222 
223  outResult->getNext( currentLevel ) = nextNode;
224  if( prevNode->getNext( currentLevel ).trySetFromTo( nextNode, outResult, NodePtr::TAG_FailIfSet ) )
225  break;
226  else
227  outResult->getNext( currentLevel ) = 0;
228  }
229 
230  currentLevel ++;
231  }
232  while( currentLevel <= resultNodeLevel
233  && !outResult->isMarkedForDeletion() ); // No point linking up remaining levels if another thread already took this node.
234 }

242 template< typename T, typename K, bool SORT_MIN_TO_MAX, U32 MAX_LEVEL, U32 PROBABILISTIC_BIAS >
243 bool ThreadSafePriorityQueue< T, K, SORT_MIN_TO_MAX, MAX_LEVEL, PROBABILISTIC_BIAS >::takeNext( T& outValue, KeyType upToPriority )
244 {
245  // Iterate through to the first unmarked node.
246 
247  NodePtr prevNode = mHead;
248  while( 1 )
249  {
250  NodePtr node;
251  readNext( prevNode, node, 0 );
252 
253  if( node == mTail )
254  return false; // End reached.
255 
256  bool priorityThresholdReached = SORT_MIN_TO_MAX
257  ? ( upToPriority >= node->getPriority() )
258  : ( upToPriority <= node->getPriority() );
259 
260  if( !priorityThresholdReached )
261  return false;
262  else
263  {
264  // Try to mark the node for deletion. Only if that succeeds, taking the
265  // node was a success and we can return. If it fails, spin and try again.
266 
267  if( node->tryMarkForDeletion() )
268  {
269  helpDelete();
270 
271  // Node is now off the list and will disappear as soon as
272  // all references held by threads (including this one)
273  // go out of scope.
274 
275  outValue = node->getValue();
276  node->clearValue();
277 
278  return true;
279  }
280  }
281  }
282 }

333 void ThreadSafePriorityQueue< T, K, SORT_MIN_TO_MAX, MAX_LEVEL, PROBABILISTIC_BIAS >::scanFromHead( NodePtr& refPrev, NodePtr& refNext, U32 level, KeyType priority )
334 {
335  // Purge dead nodes at left end of queue so
336  // we don't get stuck hitting the same node
337  // in deletable state over and over again.
338  helpDelete();
339 
340  S32 currentLevel = MAX_LEVEL - 1;
341  refPrev = mHead;
342  do
343  {
344  scan( refPrev, refNext, currentLevel, priority );
345  currentLevel --;
346  }
347  while( currentLevel >= S32( level ) );
348 }

350 template< typename T, typename K, bool SORT_MIN_TO_MAX, U32 MAX_LEVEL, U32 PROBABILISTIC_BIAS >
351 void ThreadSafePriorityQueue< T, K, SORT_MIN_TO_MAX, MAX_LEVEL, PROBABILISTIC_BIAS >::helpDelete()
352 {
353  // Clean out all the references from head.
354  // Spin over a given reference on each level until head
355  // clearly refers to a node in non-deletable state. This
356  // makes this code work cooperatively with other threads
357  // doing takeNexts on prior or later nodes while also
358  // guaranteeing that all next pointers to us will eventually
359  // disappear.
360  //
361  // Note that this is *the only place* where we will be cleaning
362  // out our lists.
363 
364  S32 level = MAX_LEVEL - 1;
365  do
366  {
367  while( 1 )
368  {
369  NodePtr ptr = mHead->getNext( level );
370  if( !ptr->isMarkedForDeletion() )
371  break;
372  else
373  {
374  NodePtr& next = ptr->getNext( level );
375  next.setTag();
376  mHead->getNext( level ).trySetFromTo( ptr, next, NodePtr::TAG_Unset );
377  AssertFatal( next->getRefCount() >= 2, "ThreadSafePriorityQueue::helpDelete() - invalid refcount" );
378  }
379  }
380 
381  level --;
382  }
383  while( level >= 0 );
384 }

465 template< typename T, typename K, bool SORT_MIN_TO_MAX, U32 MAX_LEVEL, U32 PROBABILISTIC_BIAS >
466 void* ThreadSafePriorityQueue< T, K, SORT_MIN_TO_MAX, MAX_LEVEL, PROBABILISTIC_BIAS >::Node::operator new( size_t size, S32 level )
467 {
468  if( level == -1 )
469  level = randomLevel();
470 
471  Node* node = 0;
472  while( 1 )
473  {
474  // Try to take a node from the freelist. If there's none,
475  // allocate a new one.
476 
477  if( !smFreeLists[ level ].mDestroyed )
478  node = Node::safeRead( smFreeLists[ level ].mNodes );
479 
480  if( !node )
481  {
482  node = ( Node* ) dMalloc( sizeof( Node ) + sizeof( Node* ) * level );
483  dMemset( node, 0, sizeof( Node ) );
484  node->mLevel = level;
485  node->addRef();
486  break;
487  }
488  else if( dCompareAndSwap( smFreeLists[ level ].mNodes, node, node->mNext[ 0 ] ) )
489  {
490  node->clearLowestBit();
491  break;
492  }
493  else
494  node->release(); // Other thread was quicker than us; release.
495  }
496 
497  AssertFatal( node->getRefCount() != 0, "ThreadSafePriorityQueue::new Node() - invalid refcount" );
498  AssertFatal( ( node->getRefCount() % 2 ) == 0, "ThreadSafePriorityQueue::new Node() - invalid refcount" );
499  return node;
500 }

508 {

687 template< typename T, typename K, bool SORT_MIN_TO_MAX, U32 MAX_LEVEL, U32 PROBABILISTIC_BIAS >
688 void ThreadSafePriorityQueueWithUpdate< T, K, SORT_MIN_TO_MAX, MAX_LEVEL, PROBABILISTIC_BIAS >::updatePriorities()
689 {
690  TickType currentTime = getTick();
691  U32 numNodesUpdated = 0;
692  U32 numNodesDead = 0;
693  U32 numNodesChanged = 0;
694 
695  NodePtr node;
696  while( mUpdateQueue.takeNext( node, currentTime ) )
697  {
698  numNodesUpdated ++;
699 
700  // Since we're updating nodes on the update queue only periodically,
701  // their associated values or main queue nodes may have died in the
702  // meantime. If so, we just discard them here.
703 
704  if( node->getValue().isAlive()
705  && !node->isMarkedForDeletion() )
706  {
707  KeyType newPriority = node->getValue().getPriority() + getTimeBasedPriorityBoost();
708  if( newPriority != node->getPriority() )
709  {
710  // Node is outdated. Reinsert with new priority and mark the
711  // old node for deletion.
712 
713  insert( newPriority, node->getValue() );
714  node->tryMarkForDeletion();
715  numNodesChanged ++;
716  }
717  else
718  {
719  // Node is still current. Just move to end.
720 
721  mUpdateQueue.insert( currentTime + getUpdateInterval(), node );
722  }
723  }
724  else
725  numNodesDead ++;
726  }
727 
728  #ifdef DEBUG_SPEW
729  if( numNodesUpdated )
730  Platform::outputDebugString( "[ThreadSafePriorityQueueWithUpdate] updated %i nodes (%i changed, %i dead)",
731  numNodesUpdated, numNodesChanged, numNodesDead );
732  #endif
733 }
