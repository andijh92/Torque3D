34 // Collada allows custom data to be included with many elements using the <extra>
35 // tag, followed by one or more named technique profiles. eg.
36 // <some_element>
37 // <extra>
38 // <technique profile="SOME_PROFILE">
39 // <custom_element0>value0</custom_element0>
40 // <custom_element1>value1</custom_element1>
41 // ...
42 // <technique profile="ANOTHER_PROFILE">
43 // <custom_element0>value0</custom_element0>
44 // <custom_element1>value1</custom_element1>
45 // ...
46 //
47 // This class provides an easy way to read the custom parameters into a strongly
48 // typed subclass.
49 class ColladaExtension
50 {
51  // Helper macro to simplify getting named parameters
52  #define GET_EXTRA_PARAM(param, defaultVal) \
53  get(#param, param, defaultVal)
54 
55 protected:
56  const domTechnique* pTechnique;
57 
59  template<class T> const domTechnique* findExtraTechnique(const T* element, const char* name) const
60  {
61  if (element) {
62  for (S32 iExt = 0; iExt < element->getExtra_array().getCount(); iExt++) {
63  for (S32 iTech = 0; iTech < element->getExtra_array()[iExt]->getTechnique_array().getCount(); iTech++) {
64  if (dStrEqual(element->getExtra_array()[iExt]->getTechnique_array()[iTech]->getProfile(), name))
65  return element->getExtra_array()[iExt]->getTechnique_array()[iTech];
66  }
67  }
68  }
69  return NULL;
70  }
71 
74  const domTechnique* findExtraTechnique(
75  const domCommon_color_or_texture_type_complexType::domTexture* element, const char* name) const
76  {
77  if (element && element->getExtra()) {
78  for (S32 iTech = 0; iTech < element->getExtra()->getTechnique_array().getCount(); iTech++) {
79  if (dStrEqual(element->getExtra()->getTechnique_array()[iTech]->getProfile(), name))
80  return element->getExtra()->getTechnique_array()[iTech];
81  }
82  }
83  return NULL;
84  }
85 
87  const domAny* findParam(const char* name)
88  {
89  if (pTechnique) {
90  // search the technique contents for the desired parameter
91  for (S32 iParam = 0; iParam < pTechnique->getContents().getCount(); iParam++) {
92  const domAny* param = daeSafeCast<domAny>(pTechnique->getContents()[iParam]);
93  if (param && !dStrcmp(param->getElementName(), name))
94  return param;
95  }
96  }
97  return NULL;
98  }
99 
101  template<typename T> void get(const char* name, T& value, T defaultVal)
102  {
103  value = defaultVal;
104  if (const domAny* param = findParam(name))
105  value = convert<T>(param->getValue());
106  }
107 
109  template<typename T> void get(const char* name, AnimatedElement<T>& value, T defaultVal)
110  {
111  value.defaultVal = defaultVal;
112  if (const domAny* param = findParam(name))
113  value.element = param;
114  }
115 
116 public:
117  ColladaExtension() : pTechnique(0) { }
118  virtual ~ColladaExtension() { }
119 };

159  ColladaExtension_effect(const domEffect* effect)

161  {
162  //----------------------------------
163  // <effect>
164  // MAX3D profile
165  pTechnique = findExtraTechnique(effect, "MAX3D");
166  GET_EXTRA_PARAM(double_sided, false);
167 
168  //----------------------------------
169  // <effect>.<profile_COMMON>
170  const domProfile_COMMON* profileCommon = ColladaUtils::findEffectCommonProfile(effect);
171 
172  // GOOGLEEARTH profile (same double_sided element)
173  pTechnique = findExtraTechnique(profileCommon, "GOOGLEEARTH");
174  GET_EXTRA_PARAM(double_sided, double_sided);
175 
176  //----------------------------------
177  // <effect>.<profile_COMMON>.<technique>.<blinn/phong/lambert>.<diffuse>.<texture>
178  const domCommon_color_or_texture_type_complexType* domDiffuse = ColladaUtils::findEffectDiffuse(effect);
179  const domFx_sampler2D_common_complexType* sampler2D = ColladaUtils::getTextureSampler(effect, domDiffuse);
180 
181  // Use the sampler2D to set default values for wrap/mirror flags
182  wrapU = wrapV = true;
183  mirrorU = mirrorV = false;
184  if (sampler2D) {
185  domFx_sampler2D_common_complexType::domWrap_s* wrap_s = sampler2D->getWrap_s();
186  domFx_sampler2D_common_complexType::domWrap_t* wrap_t = sampler2D->getWrap_t();
187 
188  mirrorU = (wrap_s && wrap_s->getValue() == FX_SAMPLER_WRAP_COMMON_MIRROR);
189  wrapU = (mirrorU || !wrap_s || (wrap_s->getValue() == FX_SAMPLER_WRAP_COMMON_WRAP));
190  mirrorV = (wrap_t && wrap_t->getValue() == FX_SAMPLER_WRAP_COMMON_MIRROR);
191  wrapV = (mirrorV || !wrap_t || (wrap_t->getValue() == FX_SAMPLER_WRAP_COMMON_WRAP));
192  }
193 
194  // MAYA profile
195  pTechnique = findExtraTechnique(domDiffuse ? domDiffuse->getTexture() : 0, "MAYA");
196  GET_EXTRA_PARAM(wrapU, wrapU); GET_EXTRA_PARAM(wrapV, wrapV);
197  GET_EXTRA_PARAM(mirrorU, mirrorU); GET_EXTRA_PARAM(mirrorV, mirrorV);
198  GET_EXTRA_PARAM(coverageU, 1.0); GET_EXTRA_PARAM(coverageV, 1.0);
199  GET_EXTRA_PARAM(translateFrameU, 0.0); GET_EXTRA_PARAM(translateFrameV, 0.0);
200  GET_EXTRA_PARAM(rotateFrame, 0.0);
201  GET_EXTRA_PARAM(stagger, false);
202  GET_EXTRA_PARAM(repeatU, 1.0); GET_EXTRA_PARAM(repeatV, 1.0);
203  GET_EXTRA_PARAM(offsetU, 0.0); GET_EXTRA_PARAM(offsetV, 0.0);
204  GET_EXTRA_PARAM(rotateUV, 0.0);
205  GET_EXTRA_PARAM(noiseU, 0.0); GET_EXTRA_PARAM(noiseV, 0.0);
206 
207  // FCOLLADA profile
208  if (profileCommon) {
209  pTechnique = findExtraTechnique((const domProfile_COMMON::domTechnique*)profileCommon->getTechnique(), "FCOLLADA");
210  if (pTechnique) {
211  domAny* bump = daeSafeCast<domAny>(const_cast<domTechnique*>(pTechnique)->getChild("bump"));
212  if (bump) {
213  domAny* bumpTexture = daeSafeCast<domAny>(bump->getChild("texture"));
214  if (bumpTexture) {
215  daeSIDResolver resolver(const_cast<domEffect*>(effect), bumpTexture->getAttribute("texture").c_str());
216  domCommon_newparam_type* param = daeSafeCast<domCommon_newparam_type>(resolver.getElement());
217  if (param)
218  bumpSampler = param->getSampler2D();
219  }
220  }
221  }
222  }
