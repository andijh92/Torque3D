722 // The next thing is a bit tricky. DefineEngineMethod allows to make the 'object' (=this) argument to the function
723 // implicit which presents quite an obstacle for the macro internals as the engine export system requires the
724 // name of a DLL symbol that represents an extern "C" function with an explicit first object pointer argument.
725 //
726 // Even if we ignored the fact that we don't have a guarantee how the various C++ compilers implement implicit 'this' arguments,
727 // we could still not just use a C++ method for this as then we would have to get past the C++ compiler's mangling to
728 // get to the function symbol name (let alone the fact that typing this method correctly would be tricky).
729 //
730 // So, the trick employed here is to package all but the implicit 'this' argument in a structure and then define an
731 // extern "C" function that takes the object pointer as a first argument and the struct type as the second argument.
732 // This will result in a function with an identical stack call frame layout to the function we want.
733 //
734 // Unfortunately, that still requires that function to chain on to the real user-defined function. To do this
735 // cleanly and portably, _EngineMethodTrampoline is used to unpack and jump the call from extern "C" into C++ space.
736 // In optimized builds, the compiler should be smart enough to pretty much optimize all our trickery here away.
737 
738 #define _DefineMethodTrampoline( className, name, returnType, args ) \
739  TORQUE_API EngineTypeTraits< returnType >::ReturnValueType \
740  fn ## className ## _ ## name ( className* object, _EngineMethodTrampoline< _ ## className ## name ## frame, returnType args >::Args a ) \
741  { \
742  _CHECK_ENGINE_INITIALIZED( className::name, returnType ); \
743  return EngineTypeTraits< returnType >::ReturnValue( \
744  _EngineMethodTrampoline< _ ## className ## name ## frame, returnType args >::jmp( object, a ) \
745  ); \
746  }
