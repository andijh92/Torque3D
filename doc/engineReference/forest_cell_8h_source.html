89  inline U32 _getSubCell( F32 x, F32 y ) const
90  {
91  bool left = x < ( mRect.point.x + ( mRect.extent.x / 2.0f ) );
92  bool top = y < ( mRect.point.y + ( mRect.extent.y / 2.0f ) );
93 
94  if ( left && top ) return 0;
95  if ( left && !top ) return 1;
96  if ( !left && top ) return 2;
