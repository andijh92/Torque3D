135 inline void BitMatrix::setBit(const U32 x, const U32 y)
136 {
137  AssertFatal(x < mWidth && y < mHeight, "Error, out of bounds bit!");
138 
139  U8* pRow = &mBits[y * mRowByteWidth];
140 
141  U8* pByte = &pRow[x >> 3];

161
