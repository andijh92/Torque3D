261 template< S32 NUM_DIMENSIONS, class Object >
262 void SFXWorld< NUM_DIMENSIONS, Object >::update()
263 {
264  if( !this->mReferenceObject )
265  return;
266 
267  // Get the reference center (listener) pos.
268 
269  F32 listenerPos[ NUM_DIMENSIONS ];
270  for( U32 n = 0; n < NUM_DIMENSIONS; n ++ )
271  listenerPos[ n ] = Deref( this->mReferenceObject ).getMinTrackingNode( n )->getPosition();
272 
273  // Check all current scopes.
274 
275  SFXSoundscapeManager* soundscapeMgr = SFX->getSoundscapeManager();
276  for( U32 i = 0; i < mScopeStack.size(); ++ i )
277  {
278  Scope& scope = mScopeStack[ i ];
279 
280  if( Deref( scope.mObject ).containsPoint( listenerPos ) )
281  {
282  // Listener is in the object.
283 
284  if( !scope.mSoundscape )
285  {
286  // Add the object's ambient space to the soundscape mix.
287 
288  SFXAmbience* ambience = Deref( scope.mObject ).getAmbience();
289  AssertFatal( ambience != NULL, "SFXWorld::update() - object on stack that does not have an ambient space!" );
290 
291  // Find the index at which to insert the ambient space. 0 is always
292  // the global space and each active soundscape lower down our stack
293  // increments by one.
294 
295  U32 index = 1;
296  for( U32 j = 0; j < i; ++ j )
297  if( mScopeStack[ j ].mSoundscape )
298  ++ index;
299 
300  scope.mSoundscape = soundscapeMgr->insertSoundscape( index, ambience );
301  }
302  }
303  else
304  {
305  SFXAmbience* ambience = Deref( scope.mObject ).getAmbience();
306  AssertFatal( ambience != NULL, "SFXWorld::update() - object on stack that does not have an ambient space!" );
307 
308  // Listener is neither inside object nor in its
309  // proximity zone. Kill its ambient soundscape if it
310  // has one.
311 
312  if( scope.mSoundscape )
313  {
314  soundscapeMgr->removeSoundscape( scope.mSoundscape );
315  scope.mSoundscape = NULL;
316  }
317  }
318  }
319 }

344 template< int NUM_DIMENSIONS, class Object >
345 void SFXWorld< NUM_DIMENSIONS, Object >::_onScopeIn( Object object )
346 {
347  #ifdef DEBUG_SPEW
348  Platform::outputDebugString( "[SFXWorld] Object 0x%x in scope now", object );
349  #endif
350 
351  // Get the object's ambience properties. If it has
352  // none, ignore the object.
353 
354  SFXAmbience* ambience = Deref( object ).getAmbience();
355  if( !ambience )
356  return;
357 
358  // Find where to insert the object into the stack.
359 
360  typename Vector< Scope >::iterator iter = mScopeStack.begin();
361  F32 sortValue = _getSortValue( object );
362  while( iter != mScopeStack.end() && iter->mSortValue >= sortValue )
363  ++ iter;
364 
365  // Insert the object into the stack.
366 
367  mScopeStack.insert( iter, Scope( sortValue, object ) );
368 }

372 template< S32 NUM_DIMENSIONS, class Object >
373 void SFXWorld< NUM_DIMENSIONS, Object >::_onScopeOut( Object object )
374 {
375  #ifdef DEBUG_SPEW
376  Platform::outputDebugString( "[SFXWorld] Object 0x%x out of scope now", object );
377  #endif
378 
379  // Find the object on the stack.
380 
381  S32 index = _findScope( object );
382  if( index == -1 )
383  return;
384 
385  // Remove its soundscape.
386 
387  Scope& scope = mScopeStack[ index ];
388  if( scope.mSoundscape )
389  SFX->getSoundscapeManager()->removeSoundscape( scope.mSoundscape );
390 
391  mScopeStack.erase( index );
392 }

397 F32 SFXWorld< NUM_DIMENSIONS, Object >::_getSortValue( Object object )
