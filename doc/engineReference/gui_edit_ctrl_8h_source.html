201  static void snapDelta( snappingAxis axis, snappingEdges edge, S32 offset, const RectI& bounds, Point2I& delta )
202  {
203  switch( axis )
204  {
205  case SnapVertical:
206  switch( edge )
207  {
208  case SnapEdgeMin:
209  delta.x = offset - bounds.point.x;
210  break;
211 
212  case SnapEdgeMid:
213  delta.x = offset - bounds.point.x - bounds.extent.x / 2;
214  break;
215 
216  case SnapEdgeMax:
217  delta.x = offset - bounds.point.x - bounds.extent.x + 1;
218  break;
219  }
220  break;
221 
222  case SnapHorizontal:
223  switch( edge )
224  {
225  case SnapEdgeMin:
226  delta.y = offset - bounds.point.y;
227  break;
228 
229  case SnapEdgeMid:
230  delta.y = offset - bounds.point.y - bounds.extent.y / 2;
231  break;
232 
233  case SnapEdgeMax:
234  delta.y = offset - bounds.point.y - bounds.extent.y + 1;
235  break;
236  }
237  break;
238  }
