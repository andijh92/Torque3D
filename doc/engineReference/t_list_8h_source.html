367 template<class Type> void List<Type>::operator=(const List& p)
368 {
369  if (_size >= p._size)
370  {
371  // Destroy the elements we're not going to use and copy the rest
372  while (_size != p._size)
373  popBack();
374  }
375 
376  U32 count = _size;
377  ConstIterator src = p.begin();
378  Iterator dst = begin();
379  while (count--)
380  *dst++ = *src++;
381  while (src != p.end())
382  pushBack(*src++);
383 }
