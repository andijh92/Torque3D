1023  SimObjectPtr() {}
1024  SimObjectPtr(T *ptr) { this->mReference = NULL; set(ptr); }
1025  SimObjectPtr( const SimObjectPtr& ref ) { this->mReference = NULL; set(ref.mReference); }
1026 
1027  T* getObject() const { return Parent::getPointer(); }
1028 
1029  ~SimObjectPtr() { set((WeakRefBase::WeakReference*)NULL); }
1030 
1031  SimObjectPtr<T>& operator=(const SimObjectPtr ref)
1032  {
1033  set(ref.mReference);
1034  return *this;
1035  }
1036  SimObjectPtr<T>& operator=(T *ptr)
1037  {
1038  set(ptr);
1039  return *this;
1040  }
1041 
1042  protected:
1043  void set(WeakRefBase::WeakReference * ref)
1044  {
1045  if( ref == this->mReference )
1046  return;
1047 
1048  if( this->mReference )
1049  {
1050  // Auto-delete
1051  T* obj = this->getPointer();
1052  if ( this->mReference->getRefCount() == 2 && obj && obj->isAutoDeleted() )
1053  obj->deleteObject();
1054 
1055  this->mReference->decRefCount();
1056  }
1057  this->mReference = NULL;
1058  if( ref )
1059  {
1060  this->mReference = ref;
1061  this->mReference->incRefCount();
1062  }
1063  }
1064 
1065  void set(T * obj)
1066  {
1067  set(obj ? obj->getWeakReference() : (WeakRefBase::WeakReference *)NULL);
1068  }
1069 };
