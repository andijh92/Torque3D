155  template<typename T> T* acquireAsset( const char* pAssetId )
156  {
157  // Sanity!
158  AssertFatal( pAssetId != NULL, "Cannot acquire NULL asset Id." );
159 
160  // Is this an empty asset Id?
161  if ( *pAssetId == 0 )
162  {
163  // Yes, so return nothing.
164  return NULL;
165  }
166 
167  // Find asset.
168  AssetDefinition* pAssetDefinition = findAsset( pAssetId );
169 
170  // Did we find the asset?
171  if ( pAssetDefinition == NULL )
172  {
173  // No, so warn.
174  Con::warnf( "Asset Manager: Failed to acquire asset Id '%s' as it does not exist.", pAssetId );
175  return NULL;
176  }
177 
178  // Is asset loading?
179  if ( pAssetDefinition->mAssetLoading == true )
180  {
181  // Yes, so we've got a circular loop which we cannot resolve!
182  Con::warnf( "Asset Manager: Failed to acquire asset Id '%s' as loading it involves a cyclic dependency on itself which cannot be resolved.", pAssetId );
183  return NULL;
184  }
185 
186  // Info.
187  if ( mEchoInfo )
188  {
189  Con::printSeparator();
190  Con::printf( "Asset Manager: Started acquiring Asset Id '%s'...", pAssetId );
191  }
192 
193  // Is the asset already loaded?
194  if ( pAssetDefinition->mpAssetBase == NULL )
195  {
196  // No, so info
197  if ( mEchoInfo )
198  {
199  // Fetch asset Id.
200  StringTableEntry assetId = StringTable->insert( pAssetId );
201 
202  // Find any asset dependencies.
203  typeAssetDependsOnHash::Iterator assetDependenciesItr = mAssetDependsOn.find( assetId );
204 
205  // Does the asset have any dependencies?
206  if ( assetDependenciesItr != mAssetDependsOn.end() )
207  {
208  // Yes, so show all dependency assets.
209  Con::printf( "Asset Manager: > Found dependencies:" );
210 
211  // Iterate all dependencies.
212  while( assetDependenciesItr != mAssetDependsOn.end() && assetDependenciesItr->key == assetId )
213  {
214  // Info.
215  Con::printf( "Asset Manager: > Asset Id '%s'", assetDependenciesItr->value );
216 
217  // Next dependency.
218  assetDependenciesItr++;
219  }
220  }
221  }
222 
223  // Flag asset as loading.
224  pAssetDefinition->mAssetLoading = true;
225 
226  // Generate primary asset.
227  pAssetDefinition->mpAssetBase = mTaml.read<T>( pAssetDefinition->mAssetBaseFilePath );
228 
229  // Flag asset as finished loading.
230  pAssetDefinition->mAssetLoading = false;
231 
232  // Did we generate the asset?
233  if ( pAssetDefinition->mpAssetBase == NULL )
234  {
235  // No, so warn.
236  Con::warnf( "Asset Manager: > Failed to acquire asset Id '%s' as loading the asset file failed to return the asset or the correct asset type: '%s'.",
237  pAssetId, pAssetDefinition->mAssetBaseFilePath );
238  return NULL;
239  }
240 
241  // Increase loaded count.
242  pAssetDefinition->mAssetLoadedCount++;
243 
244  // Info.
245  if ( mEchoInfo )
246  {
247  Con::printf( "Asset Manager: > Loading asset into memory as object Id '%d' from file '%s'.",
248  pAssetDefinition->mpAssetBase->getId(), pAssetDefinition->mAssetBaseFilePath );
249  }
250 
251  // Set ownership by asset manager.
252  pAssetDefinition->mpAssetBase->setOwned( this, pAssetDefinition );
253 
254  // Is the asset internal?
255  if ( pAssetDefinition->mAssetInternal )
256  {
257  // Yes, so increase internal loaded asset count.
258  if ( ++mLoadedInternalAssetsCount > mMaxLoadedInternalAssetsCount )
259  mMaxLoadedInternalAssetsCount = mLoadedInternalAssetsCount;
260  }
261  else
262  {
263  // No, so increase external loaded assets count.
264  if ( ++mLoadedExternalAssetsCount > mMaxLoadedExternalAssetsCount )
265  mMaxLoadedExternalAssetsCount = mLoadedExternalAssetsCount;
266  }
267  }
268  else if ( pAssetDefinition->mpAssetBase->getAcquiredReferenceCount() == 0 )
269  {
270  // Info.
271  if ( mEchoInfo )
272  {
273  Con::printf( "Asset Manager: > Acquiring from idle state." );
274  }
275  }
276 
277  // Set acquired asset.
278  T* pAcquiredAsset = dynamic_cast<T*>( (AssetBase*)pAssetDefinition->mpAssetBase );
279 
280  // Is asset the correct type?
281  if ( pAcquiredAsset == NULL )
282  {
283  // No, so warn.
284  Con::warnf( "Asset Manager: > Failed to acquire asset Id '%s' as it was not the required asset type: '%s'.", pAssetId, pAssetDefinition->mAssetBaseFilePath );
285  return NULL;
286  }
287 
288  // Acquire asset reference.
289  pAcquiredAsset->acquireAssetReference();
290 
291  // Info.
292  if ( mEchoInfo )
293  {
294  Con::printf( "Asset Manager: > Finished acquiring asset. Reference count now '%d'.", pAssetDefinition->mpAssetBase->getAcquiredReferenceCount() );
295  Con::printSeparator();
296  }
297 
298  return pAcquiredAsset;
299  }

302  template<typename T> T* acquireAsPrivateAsset( const char* pAssetId )
303  {
304  // Acquire the asset normally.
305  T* pAsset = acquireAsset<T>( pAssetId );
306 
307  // Finish if the asset was not acquired.
308  if ( pAsset == NULL )
309  return NULL;
310 
311  // Clone the asset.
312  T* pAssetClone = dynamic_cast<T*>( pAsset->clone() );
313 
314  // Sanity!
315  AssertFatal( pAssetClone != NULL, "acquireAsPrivateAsset() - Failed to clone asset type." );
316 
317  // Release the public asset.
318  releaseAsset( pAssetId );
319 
320  // Add as a private asset.
321  addPrivateAsset( pAssetClone );
322 
323  return pAssetClone;
324  }
