197  bool isSilhouetteEdge( U32 edgeIndex ) const
198  {
199  AssertFatal( edgeIndex < this->mPolyhedron->getNumEdges(), "SilhouetteExtractorBaseOrtho::isSilhouetteEdge - Index out of range!" );
200 
201  const typename Polyhedron::EdgeType& edge = this->mPolyhedron->getEdges()[ edgeIndex ];
202 
203  const U32 face0 = edge.face[ 0 ];
204  const U32 face1 = edge.face[ 1 ];
205 
206  // Not a silhouette if both planes are facing the same way.
207 
208  if( mSign( mFaceDotProducts[ face0 ] ) == mSign( mFaceDotProducts[ face1 ] ) )
209  return false;
210 
211  // Find out which face is the front facing one. Since we expect normals
212  // to be pointing inwards, this means a reversal of the normal back facing
213  // test and we're looking for a normal facing the *same* way as our projection.
214 
215  const U32 frontFace = mFaceDotProducts[ face0 ] > 0.f ? face0 : face1;
216  if( mFaceDotProducts[ frontFace ] <= 0.f )
217  return false; // This face or other face is perpendicular to us.
218 
219  return true;
220  }

225 template< typename Base >
226 struct SilhouetteExtractorImpl : public Base
227 {
228  typedef typename Base::PolyhedronType PolyhedronType;
229 
230  SilhouetteExtractorImpl( const PolyhedronType& polyhedron )
231  : Base( polyhedron ) {}
232 
233  U32 extractSilhouette( U32* outIndices, U32 maxOutIndices ) const
234  {
235  // First, find the silhouette edges. We do this with a brute-force
236  // approach here. This can be optimized (see "Silhouette Algorithms" by Bruce Gooch, Mark
237  // Hartner, and Nathan Beddes).
238 
239  U32 numSilhouetteEdges = 0;
240  const U32 numTotalEdges = this->mPolyhedron->getNumEdges();
241  const typename PolyhedronType::EdgeType* edges = this->mPolyhedron->getEdges();
242  FrameTemp< const typename PolyhedronType::EdgeType* > silhouetteEdges( numTotalEdges );
243 
244  for( U32 i = 0; i < numTotalEdges; ++ i )
245  if( this->isSilhouetteEdge( i ) )
246  silhouetteEdges[ numSilhouetteEdges ++ ] = &edges[ i ];
247 
248  // Allow this to happen rather than asserting as projection-based silhouettes
249  // may fail.
250  if( numSilhouetteEdges < 3 )
251  return 0;
252 
253  // Now walk the edge list and find the edges that are connected
254  // with each other. From this information, emit the silhouette
255  // polygon.
256 
257  U32 idx = 0;
258 
259  if( idx >= maxOutIndices )
260  return 0;
261  outIndices[ idx ++ ] = silhouetteEdges[ 0 ]->vertex[ 1 ];
262 
263  U32 currentIndex = silhouetteEdges[ 0 ]->vertex[ 1 ];
264  U32 currentEdge = 0;
265 
266  for( U32 i = 1; i < numSilhouetteEdges; ++ i )
267  {
268  // Find edge that continues on from the current vertex.
269  for( U32 n = 0; n < numSilhouetteEdges; ++ n )
270  {
271  // Skip current edge.
272  if( n == currentEdge )
273  continue;
274 
275  if( silhouetteEdges[ n ]->vertex[ 0 ] == currentIndex )
276  currentIndex = silhouetteEdges[ n ]->vertex[ 1 ];
277  else if( silhouetteEdges[ n ]->vertex[ 1 ] == currentIndex )
278  currentIndex = silhouetteEdges[ n ]->vertex[ 0 ];
279  else
280  continue;
281 
282  if( idx >= maxOutIndices )
283  return 0;
284 
285  currentEdge = n;
286  outIndices[ idx ++ ] = currentIndex;
287  break;
288  }
289  }
290 
291  return idx;
292  }
293 };
