146  // isLeft(): tests if a point is Left|On|Right of an infinite line.
147  // Input: three points P0, P1, and P2
148  // Return: >0 for P2 left of the line through P0 and P1
149  // =0 for P2 on the line
150  // <0 for P2 right of the line
151  // See: the January 2001 Algorithm on Area of Triangles
152  inline F32 isLeft( const Point3F &P0, const Point3F &P1, const Point3F &P2 )
153  {
154  return (P1.x - P0.x)*(P2.y - P0.y) - (P2.x - P0.x)*(P1.y - P0.y);
155  }
