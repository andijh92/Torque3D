27 // [tom, 1/29/2007] ZipStatFilter allows us to track CRC and uncompressed size
28 // on the fly. This is necessary when dealing with compressed files as the
29 // CRC must be of the uncompressed data.
30 //
31 // The alternative would be to compress the files when updating the zip file,
32 // but this could potentially cause ZipArchive::closeArchive() to take a long
33 // time to complete. With compression done on the fly the time consuming code
34 // is pushed out to when writing to files, which is significantly easier to
35 // do asynchronously.

112  virtual bool _write(const U32 numBytes, const void *buffer)
113  {
114  if(! mStream->write(numBytes, buffer))
115  return false;
116 
117  mCD->mUncompressedSize += numBytes;
118  mCD->mCRC32 = CRC::calculateCRC(buffer, numBytes, mCD->mCRC32);
119 
120  return true;
121  }
122 
123  virtual bool _read(const U32 numBytes, void *buffer)
124  {
125  if(! mStream->read(numBytes, buffer))
126  return false;
127 
128  mCD->mUncompressedSize += numBytes;
129  mCD->mCRC32 = CRC::calculateCRC(buffer, numBytes, mCD->mCRC32);
130 
131  return true;
132  }
133 
134 public:
135  ZipStatFilter() : mStream(NULL), mCD(NULL) {}
136  ZipStatFilter(CentralDir *cd) : mStream(NULL), mCD(cd) {}
137  virtual ~ZipStatFilter()
138  {
139  detachStream();

141 
142  virtual bool attachStream(Stream *stream)
143  {
144  if(mCD == NULL)
145  return false;
146 
147  mStream = stream;
148  mCD->mUncompressedSize = 0;
149  mCD->mCRC32 = CRC::INITIAL_CRC_VALUE;
150  return true;
151  }
152 
153  virtual void detachStream()
154  {
155  if(mStream == NULL)
156  return;
157 
158  // Post condition the CRC
159  mCD->mCRC32 ^= CRC::CRC_POSTCOND_VALUE;
160  mStream = NULL;
161  }
